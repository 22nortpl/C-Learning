#include <stdio.h>

int main(void)
{
	printf("2 + 3 = %d\n", 2 + 3);
	printf("2 - 3 = %d\n", 2 - 3);
	printf("2 * 3 = %d\n", 2 * 3);
	printf("2 / 3 = %d\n", 2 / 3);
	return 0;
}

결과: 
2 + 3 = 5
2 - 3 = -1
2 * 3 = 6
2 / 3 = 0
